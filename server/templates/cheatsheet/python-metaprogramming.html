Debuging via Decorator

# Using print is ugly.
>>> def spam1():
...   print spam1.__name__
...   print 'SPAM_1'
... 
>>> def spam2():
...   print spam2.__name__
...   print 'SPAM_2'
... 
>>> spam1()
spam1
SPAM_1
>>> spam2()
spam2
SPAM_2

# Using decorator
>>> def debug(func):
...   def wrapper(*args, **kwargs):
...     print func.__name__
...     return func(*args, **kwargs)
...   return wrapper
... 
>>> @debug
... def spam():
...   print 'SPAM'
... 
>>> spam()
spam
SPAM
