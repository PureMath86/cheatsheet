>>> class count(object):
...   def __init__(self,n):

What "RES = yield from EXP" actually do?

>>> def subgen():

...   _i = iter(EXP)

Implement Simple round-robin with blocking

# skill: using task and wait queue
from collections import deque
from select import select
import socket

tasks = deque()
w_read = {}
w_send = {}

def run():
   while any([tasks,w_read,w_send]):
      while not tasks:
         # polling tasks
         can_r,can_s,_ = select(
               w_read,w_send,[])
         for _r in can_r:
            tasks.append(w_read.pop(_r))
         for _w in can_s:
            tasks.append(w_send.pop(_w))
      try:
         task = tasks.popleft()
         why,what = next(task)
         if why == 'recv':
            w_read[what] = task 
         elif why == 'send':
            w_send[what] = task 
         else:
            raise RuntimeError
      except StopIteration:
         pass

def server():
   host = ('localhost',5566)
   sock = socket.socket(
         socket.AF_INET,
         socket.SOCK_STREAM)
   sock.setsockopt(
         socket.SOL_SOCKET,
         socket.SO_REUSEADDR,1)
   sock.bind(host)
   sock.listen(5)
   while True:
      # tell scheduler want block
      yield 'recv', sock 
      conn,addr = sock.accept()
      tasks.append(client_handler(conn))

def client_handler(conn):
   while True:
      # tell scheduler want block
      yield 'recv', conn
      msg = conn.recv(1024)
      if not msg:
         break
      # tell scheduler want block
      yield 'send', conn
      conn.send(msg)
   conn.close()

tasks.append(server())
run()

What @contextmanager actually doing?

# ref: PyCon 2014 - David Beazley
# define a context manager class
class GeneratorCM(object):
   def __init__(self,gen):
      self._gen = gen
   def __enter__(self):
      return next(self._gen)

   def __exit__(self, *exc_info):
      try:
         if exc_info[0] is None:
            next(self._gen)
         else:
            self._gen.throw(*exc_info)
         raise RuntimeError
      except StopIteration:
         return True
      except:
         raise
            
# define a decorator
def contextmanager(func):
   def run(*a, **k):
      return GeneratorCM(func(*a, **k))
   return run

# example of context manager
@contextmanager
def mylist():
   try:
      l=[1,2,3,4,5]
      yield l
   finally:
      print "exit scope"

with mylist() as l:
   print l

class Task(object):
   def __init__(self,gen):

   def step(self, val=None):
      # Purpose: run the next yield
      try:
         fut = self._gen.send(val)
         fut.add_done_callback(self._wkup)

   def _wkup(self, fut):

      # "send result" to generator 
      self.step(res)

Do some task after some task done

# ex.py
# wrapping generator with "Task"
class Task(object):
   def __init__(self,gen):
      self._gen = gen

   def step(self, val=None):
      # Purpose: run the next yield
      try:
         fut = self._gen.send(val)
         fut.add_done_callback(self._wakeup)
      except StopIteration:
         pass
   def _wakeup(self, fut):
      # Purpose: handling result
      res = fut.result()
      self.step(res)

# Demo how to use "Task"
import concurrent.futures as cf
import time
pool = cf.ThreadPoolExecutor(8)

def func(x,y):
   return x + y

def do_func(x,y):
   res = yield pool.submit(func,x,y)
   print("Got:", res)

# brute-force solution
def after_bf(delay,gen):
   yield pool.submit(time.sleep, delay)
   result = None
   try:
      while True:
         fut = gen.send(result)
         result = yield fut
   except StopIteration:
      pass
   
# elegant solution
def after_e(delay,gen):
   # do some task like init
   yield pool.submit(time.sleep, delay)
   # do exactly task after some task
   yield from gen

>>> from ex import Task
>>> from ex import do_func
>>> from ex import after_bf
>>> from ex import after_e
>>> Task(after_bf(5,do_func(2,3))).step()
>>> Got: 5

>>> Task(after_e(5,do_func(2,3))).step()
