p and puts

ker

=>:ker

BEGIN and END block

# run BEGIN block before main
BEGIN {
  puts 'begin ruby script'
}
# run END block after main
END {
  puts 'end ruby script'
}
# main script
puts 'ruby script main'

# bash> ruby ex.rb
# begin ruby script
# ruby script main
# end ruby script

Basic File IO

# basic read write
>> f = File.open('test','w')
=>#<File:test>
>> f.write('Hello World')
=>11
>> f.close
=>nil
>> f = File.open('test','r')
=>#
>> f.read
=>"Hello World"

Check File object closed?

>> f = File.open('test','r')
=>#<File:test>
>> f.closed?
=>false
>> f.close
=>nil
>> f.closed?
=>true

Auto Close File Object

>> File.open('test','w') do |file|
..   file.write "Hello World"
.. end
=>11

Check File Attributes

# check file exist
>> File.exist? '/etc/passwd'
=>true
# check readable
>> File.readable? '/etc/passwd'
=>true
# check writable
>> File.writable? '/etc/passwd'
=>false
# check symbolic link
>> File.symlink? '/etc/passwd'
=>false
# check directory
>> File.directory? '/usr'
=>true
