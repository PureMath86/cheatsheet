Context Manager in Ruby

# python context manager in ruby
require 'socket'
include Socket::Constants

host = "0.0.0.0"
port = 9527 

def with port, host
  begin
    s = Socket.new(AF_INET,
                  SOCK_STREAM, 0)
    s.setsockopt(SOL_SOCKET, 
                 SO_REUSEADDR, true)
    addr = Socket.sockaddr_in(port, host)
    s.bind(addr)
    s.listen(10)
    yield s
  ensure
    if not s.closed?
      s.close
    end
  end
end

with port, host do |s|
  loop do
    conn, addr = s.accept
    msg = conn.gets
    conn.puts msg
    conn.close
  end
end
# bash> nc localhost 9527
