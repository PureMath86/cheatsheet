Basic Usage of Reference and Pointer

#include <iostream>
using namespace std;

int main(int argc, char* argv[])
{
   int a = 5566;
   int *p_a_addr = &a;
   int &r_a_addr = a;

   /* value of a*/
   cout << a << endl;
   /* address of a */
   cout << &a <<endl;
   /* address of a */
   cout << p_a_addr << endl;
   /* address of pointer */
   cout << &p_a_addr <<endl;
   /* value of a */
   cout << r_a_addr << endl;
   /* address of a */
   cout << &r_a_addr << endl;
   return 0;
}

/* 
output:

5566
0x7fff586ef6ec
0x7fff586ef6ec
0x7fff586ef6e0
5566
0x7fff586ef6ec
*/

Lambda Function

/* Lambda since C++11 */
#include <iostream>
using namespace std;

int main(int argc, char *argv[])
{
   int a = 8;
   int sa = 0;

   /* lambda square(since c++11) */
   sa = [](int x) {return x*x;}(a);
   cout << "a = " << a << endl;
   cout << "square a = "
        << sa << endl;
   
   return 0;
}
/*
bash> g++ ex.cpp --std=c++11
output:

a = 8
square a = 64
*/

Different between Pointer and Reference

#include <iostream>
using namespace std;

int main(int argc, char *argv[])
{
   int a = 5566;
   int b = 9527;
   int *ptr1, *ptr2;
   int &ref1 = a; 
   int &ref2 = b;

   ptr1 = &a;
   ptr2 = &b;
   cout << "b value will not change" 
        << endl;
   cout << "val b = "
        << b  << endl;
   ptr2 = &a;
   cout  << "val b = "
         << b  << endl;

   cout  << "b value will change"
         << endl;
   cout  << "val b = "
         << b  << endl;
   ref2 = a;
   cout  << "val b = "
         << b  << endl;
   
   return 0;
}

/* 
output:

b value will not change
val b = 9527
val b = 9527
b value will change
val b = 9527
val b = 5566
*/

Call by "Reference"

#include <iostream>
using namespace std;

void spam_ptr(int *ptr)
{
   cout << "addr = " 
        << ptr << endl; 
   cout << "val = "
        << *ptr << endl;
}

void spam_ref(int &ref)
{
   cout << "addr = "
        << &ref << endl;
   cout << "val = "
        << ref <<endl;
}

int main(int argc, char *argv[])
{
   int var = 5566;
   spam_ptr(&var);
   spam_ref(var);
   
   return 0;
}
/*
output:

addr = 0x7fff53c6c74c
val = 5566
addr = 0x7fff53c6c74c
val = 5566
*/
