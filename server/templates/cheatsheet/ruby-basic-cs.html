Ruby Comments

# donnot use block comment
# bad
=begin
- Hello
- World
=end

# good
# - Hello
# - World

Ruby use ";"

# donnot use ";" between two exp
# bad
puts "Hello"; puts "World"
# good 
puts "Hello"
puts "World"

# prefer use ";" with nobody
def spam_method; end
