Simple Echo TCPServer

require 'socket'

port = 9527

server = TCPServer.new port
loop do
  client = server.accept
  msg = client.gets
  client.puts msg
  client.close
end

Simple Echo Linux TCP Socket Server

require 'socket'
include Socket::Constants

host = '0.0.0.0'
port = 9527

s = Socket.new(AF_INET, 
               SOCK_STREAM, 0)
s.setsockopt(SOL_SOCKET, 
             SO_REUSEADDR, true)
saddr = Socket.sockaddr_in(port, host)
s.bind(saddr)
s.listen(5)

loop do
  conn, addr = s.accept
  msg = conn.gets
  conn.puts msg
  conn.close
end
